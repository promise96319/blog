配置 nginx #

nginx
server {
    listen 443 ssl;
    server_name example.com;
    ssl_certificate /path/to/fullchain.cer;
    ssl_certificate_key /path/to/example.com.key;
    ssl_protocols TLSv1 TLSv1.1 TLSv1.2;
    ssl_ciphers HIGH:!aNULL:!MD5;
    location / {
        proxy_pass http://localhost:8080;
    }
}
